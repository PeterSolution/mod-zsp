#Add to scripted triggers if more levels of hyper inflation is added
VEN_inflation_clear_all = {
	remove_ideas = {
		VEN_inflation1
		VEN_inflation2
		VEN_inflation3
		VEN_inflation4
		VEN_inflation5
	}
}

#Important to put HIGHEST first
VEN_inflation_level_up = {
	# level 5 not relevant. Cannot go above.
	custom_effect_tooltip = CHI_inflation_up_tt
	if = {
		limit = { has_idea = VEN_inflation4 }
		swap_ideas = {
			remove_idea = VEN_inflation4
			add_idea = VEN_inflation5
		}
	}
	else_if = {
		limit = { has_idea = VEN_inflation3 }
		swap_ideas = {
			remove_idea = VEN_inflation3
			add_idea = VEN_inflation4
		}
	}
	else_if = {
		limit = { has_idea = VEN_inflation2 }
		swap_ideas = {
			remove_idea = VEN_inflation2
			add_idea = VEN_inflation3
		}
	}
	else_if = {
		limit = { has_idea = VEN_inflation1 }
		swap_ideas = {
			remove_idea = VEN_inflation1
			add_idea = VEN_inflation2
		}		
	}
	else_if = {
		limit = { NOT = { has_idea = VEN_inflation1 } }
		add_ideas = VEN_inflation4
	}
}

#Important to put LOWEST first
VEN_inflation_level_down = {
custom_effect_tooltip = CHI_inflation_down_tt
	if = {
		limit = { has_idea = VEN_inflation1 }
		#Makes the tooltip clearer if player looks at the effect early in the game
		effect_tooltip = {
				remove_idea = VEN_inflation1
		}
		hidden_effect = { remove_ideas = VEN_inflation1 }
	}
	if = {
		limit = { has_idea = VEN_inflation2 }
		swap_ideas = {
			remove_idea = VEN_inflation2
			add_idea = VEN_inflation1
		}
	}
	if = {
		limit = { has_idea = VEN_inflation3 }
		swap_ideas = {
			remove_idea = VEN_inflation3
			add_idea = VEN_inflation2
		}
	}
	if = {
		limit = { has_idea = VEN_inflation4 }
		swap_ideas = {
			remove_idea = VEN_inflation4
			add_idea = VEN_inflation3
		}
	}
	if = {
		limit = { has_idea = VEN_inflation5 }
		swap_ideas = {
			remove_idea = VEN_inflation5
			add_idea = VEN_inflation4
		}
	}
}
